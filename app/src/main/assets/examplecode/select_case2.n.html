பெயர்_type = ["எழி","சரண்","சரவணன்"]
பாலினம்_type = ["ஆண்","பெண்","ஒருவண்"]

count = 0
@( count < 3 ) வரை
பெயர் = எடு(பெயர்_type,count)
பாலினம் = எடு(பாலினம்_type,count)

# no break statements are required. fall-through 
# behavior is not supported
@( பாலினம் ) தேர்ந்தெடு
@( "ஆண்" ) தேர்வு
	பதிப்பி "Mr / வணக்கம் செல்வன். ", பெயர்	
    இது = "ஆண்"
@( "பெண்" )  தேர்வு 
	பதிப்பி "Miss / வணக்கம் செல்வி. ", பெயர்	
    இது = "பெண்"
ஏதேனில்
    பதிப்பி "Hallo / வணக்கம் ", பெயர்	
    இது = பாலினம்
முடி
assert( எடு(பாலினம்_type,count) == இது )
count = count+1
