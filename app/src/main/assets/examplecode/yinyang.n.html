    turtle_fill(True)
    turtle_circle(radius/2., 180)
    turtle_circle(radius, 180)
    turtle_left(180)
    turtle_circle( -1*radius/2.0 , 180 )
    turtle_color(color1)
    turtle_fill(True)
    turtle_color(color2)
    turtle_left(90)
    turtle_up()
    turtle_forward(radius*0.375)
    turtle_right(90)
    turtle_down()
    turtle_circle(radius*0.125)
    turtle_left(90)
    turtle_fill(False)
    turtle_up()
    turtle_backward(radius*0.375)
    turtle_down()
