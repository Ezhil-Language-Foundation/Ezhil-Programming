# eval function
val = eval("eval(\"sqrt(5)\") - sqrt(4)")
assert( abs(val - 0.2360679774 ) < 1e-6 )

val = eval("sqrt(5) - sqrt(4)")
assert( abs( val - 0.2360679774997898  ) < 1e-6 )
