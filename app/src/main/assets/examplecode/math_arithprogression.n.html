    மு = 0
    @( டி = 0, டி < len(x), டி = டி+1 )  ஆக
               மு  =  மு  + x[டி] 
    முடி

    பின்கொடு மு

# 5 இல் தொடங்கி 7படிகளால் 61 வரை உள்ள எண் தொடர்
தொடக்கம்=5; படி=7; 
எண்_தொடர் = range(5,61+1,7)

# கூட்டு_சூத்திரம்
அளவெண்=(61-5)/7+1;
கூட்டு_சூத்திரம் = (எண்_தொடர்[0] + எண்_தொடர்[-1])*(அளவெண்/2.0)

பதிப்பி "எண் தொடர்", எண்_தொடர்
பதிப்பி "கூட்டு =>",  கூட்டு (  எண்_தொடர் ) 
பதிப்பி "கூட்டு சூத்திரம் => ", கூட்டு_சூத்திரம்
assert( கூட்டு (  எண்_தொடர் )  == கூட்டு_சூத்திரம் )
