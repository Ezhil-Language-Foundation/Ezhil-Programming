# பெருக்கல் வாய்ப்பாடு - 1 முதல் 10, 12 வரை
# இரண்டு 'வரை' வாக்கியத்தால் பயன்படுத்தி  செய்யலாம்
அ = 1
@( அ <= 10 ) வரை
    ஆ = 1
    பதிப்பி  "################################"
    பதிப்பி  "வாய்ப்பாடு ",  அ , "- 1 முதல் 12 வரை"
    @( ஆ  <= 12 ) வரை
         பதிப்பி  அ , "*" , ஆ ,  " = ",  ( ஆ*அ )
         ஆ = ஆ + 1
    முடி
    அ =  அ + 1
