# sum of digits of a number
# எண்ணிக்கையிலான இலக்கங்களின் தொகை
# Ref: http://rosettacode.org/wiki/Sum_digits_of_an_integer

நிரல்பாகம் எண்_கூட்டல்( எண் )
  தொகை = 0
  @( எண் > 0 ) வரை
     d = எண்%10;
     பதிப்பி "digit = ",d
     எண் = (எண்-d)/10;
     தொகை  = தொகை  + d
  முடி
  பின்கொடு தொகை 
முடி
