## sort descending 
நிரல்பாகம்  bubble_sort ( n )
  i = 0;
  @( i < len(n) ) வரை
     j = i;
     @( j < len(n) ) வரை
              @( n[j] > n[i] ) ஆனால்
              	  tmp = n[j];
               	  n[j] = n[i];
		  n[i] = tmp;
              முடி 
              j = j + 1 
    முடி
    i = i + 1
  முடி
  # comment .1
  # அணிகள் குறிப்புகளால் பறிமாரப்படுகின்றன 
  # (arrays are passed by pointers)
  பின்கொடு  n; # not necessary
முடி

பதிப்பி "un-sorted => ",x
