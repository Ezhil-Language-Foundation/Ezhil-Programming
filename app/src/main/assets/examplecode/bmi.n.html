நிரல்பாகம் bmi(wt,hFt,hIn)

  ht2 = ((hFt*12+hIn)^2);
  
  பின்கொடு  (wt/ht2)*703;
முடி

# < 25
பதிப்பி "healthy = ", bmi(150,5,5)

# >= 25
பதிப்பி "overweight =", bmi(170,5,5)

# >= 30 
பதிப்பி "obese =", bmi(190,5,5)
